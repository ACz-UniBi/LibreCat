LibreCat::Worker::Mailer a worker for sending mails

NAME

LibreCat::Worker::Mailer - a worker for sending mails

SYNOPSIS

    use LibreCat::Worker::Mailer;

    my $mailer = LibreCat::Worker::Mailer->new;
    $mailer->work({
        to => 'me',
        from => 'system_mailer',
        subject => 'important',
        body => 'hello world',
    });
