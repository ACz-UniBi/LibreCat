LibreCat::Worker::DataCite

a worker for registering and minting at DataCite

NAME

LibreCat::Worker::DataCite - a worker for registering and minting at DataCite

SYNOPSIS

    use LibreCat::Worker::DataCite;

    my $registry = LibreCat::Worker::DataCite->new(user => 'me', password => 'secret');

    $registry->work({
        doi          => '...' ,
        landing_url  => '...' ,
        datacite_xml => '...' ,
    })

    # or call them separately
    $registry->metadata()
    $registry->mint('')

CONFIGURATION

user

Required.

password

Required.
