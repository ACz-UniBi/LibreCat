LibreCat::Daemon base class for librecat daemons

NAME

LibreCat::Daemon - base class for librecat daemons

SYNOPSIS

    package LibreCat::Cmd::mydaemon;

    use parent 'LibreCat::Daemon';

    sub daemon {
        sub {
            while (1) {
                log "hard at work... ";
                sleep 5;
            }
        };

    1;

COMMAND LINE USAGE

    librecat mydaemon start --workers 10 --supervise
    librecat mydaemon stop --workers 10
